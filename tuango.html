#q9

perf_cals <- function(sms, intro){
  people = 250902
  perc_mail = mean(sms$mailto)
  response_number = perc_mail * people
  dat = filter(sms, mailto == TRUE)
  rep_rate = mean(dat$buyer == "yes")
  nr_resp = response_number * rep_rate
  total_cost = cost_of_offer * response_number
  profit = round((margin_on_revenue * nr_resp- total_cost),2)
  ROME = round(profit / total_cost,5) * 100
  print(paste0(intro, ' profit is: ', profit,2))
  print(paste0(intro, ' ROME is: ', ROME,' %'))
  return(data.frame(profit, ROME))
}
