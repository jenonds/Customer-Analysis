## return value of CLV for each week 
## for every experiment(different emails received), variables are churn rate, average revenue of subscribers and unsubscribers
clv <- function(churn, revenue_s, revenue_ns, week) {
  retention_s <- c(1, cumprod(1 - churn[1:(week-1)])) 
  ###churn rate's definition is different from homealarms case
  #retention_ns <- cumsum(c(0, churn*retention_s)[1:week])
  retention_ns <- 1 - retention_s
  revenue_s_t <- revenue_s * retention_s
  revenue_ns_t <- revenue_ns * retention_ns
  revenue_t <- revenue_s_t + revenue_ns_t
  cos_t <- revenue_t * cogs
  revenues <- revenue_t - cos_t
  PV_r <- revenues / (1 + week_discount)^discount_period
  CLV <- cumsum(PV_r)
  df <- data.frame(churn, revenue_s, revenue_ns, retention_s, retention_ns, discount_period, revenue_s_t, revenue_ns_t, revenue_t, cos_t, revenues, PV_r, CLV)
  return(CLV)
  #return(PV_r)

}

#Answer Q5 One problem is the promotion of products is sent from one department, which decreases the experiments credibility in determining the effects of promotion given in real life, it is unlikely to receive one type of promotion. The adjustment could be have a combination of two to three departments sending out promotion in a regular pattern. Second problem could be some bias in lack of no controlled trail of 0-email frequency. The research is conducted based on the assumption that email is good source of advertisement. I strongly recommend use a compared experiment of no-email trail to certify that email promotes sale. A third problem could be the source of volunteers participating in the experiment. There is no evidence showing they are equally neutral to the brand. Although it is said the volunteers are randomly distributed to each group, I recommend pick up people who never heard of the brand.
